            THEN CONCAT(ROUND((( DATA_LENGTH + INDEX_LENGTH) / 1024),2), "KB")        
        ELSE CONCAT((( DATA_LENGTH + INDEX_LENGTH)), "Byte") 
    END AS `size`
FROM  
    information_schema.TABLES
